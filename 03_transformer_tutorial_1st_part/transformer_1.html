def get_positional_encoding(max_seq_len, embed_dim):
    # positional encoding
    # embed_dim: 
    # max_seq_len: 
    positional_encoding = np.array([
        [pos / np.power(10000, 2 * i / embed_dim) for i in range(embed_dim)]
        if pos != 0 else np.zeros(embed_dim) for pos in range(max_seq_len)])
    positional_encoding[1:, 0::2] = np.sin(positional_encoding[1:, 0::2])  # dim 2i 2
    positional_encoding[1:, 1::2] = np.cos(positional_encoding[1:, 1::2])  # dim 2i+1 2
    # , 
    # denominator = np.sqrt(np.sum(positional_encoding**2, axis=1, keepdims=True))
    # positional_encoding = positional_encoding / (denominator + 1e-8)
    return positional_encoding
